//! [atanh]
---- simd
<- pf =          (1, 2, -1, 0.5) + i * (0, 1, -1, 3)
-> bs::atanh(pf) = (inf, 0.402359, -0.402359, 0.0489361) + i * (-nan, 1.33897, -1.01722, 1.2564)
---- scalar
<- xf =                   0 + i * 1
-> bs::atanh(xf) =        0 + i * 0.785398
<- yf =                   2
-> bs::atanh(yf) =        nan
-> bs::cmplx::atanh(yf) = 0.549306 + i * 1.5708
