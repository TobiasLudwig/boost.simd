//! [acosh]
---- simd
<- pf =          (1, 2, -1, 0.5) + i * (0, 1, -1, 3)
-> bs::acosh(pf) = (-0, 1.46935, 1.06128, 1.83019) + i * (0, 0.507356, -2.23704, 1.41379)
---- scalar
<- xf =                   0 + i * 1
-> bs::acosh(xf) =        0.881374 + i * 1.5708
<- yf =                   -2
-> bs::acosh(yf) =        -nan
-> bs::cmplx::acosh(yf) = 1.31696 + i * 0
