---- simd
<- pf =          (1, 2, -1, 0.5) + i * (0, 1, -1, 3)
-> bs::cosh(pf) = (1.54308, 2.03272, 0.83373, -1.11634) + i * (0, 3.0519, 0.988898, 0.073537)
---- scalar
<- xf =                   0 + i * 1
-> bs::cosh(xf) =        0.540302 + i * 0
<- yf =                   2
-> bs::cosh(yf) =        3.7622
