//! [sqrt]
---- simd
<- pf =           (1, 2, -1, 0.5) + i * (0, 1, -1, 3)
-> bs::sqrt(pf) = (1, 1.45535, 0.45509, 1.33067) + i * (0, 0.343561, -1.09868, 1.12725)
---- scalar
<- xf =                  0 + i * 3.14159
-> bs::sqrt(xf) =        1.25331 + i * 1.25331
<- yf =                  -1
-> bs::sqrt(yf) =        -nan
-> bs::cmplx::sqrt(yf) = 0 + i * 1
