//! [pow]
---- simd
 <- pf =              (0, 2, -1, 0.5) + i * (0, 1, -1, 3)
 <- qf =              (0, 1, -1, 3) + i * (0, 2, -1, 0.5)
 -> bs::pow(pf, qf) = (1, -0.425894, -0.0284751, 0.845227) + i * (0, 0.77537, 0.0606697, -13.9053)
---- scalar
 xf =                          -1 + i * 1.5
 yf =                          4 + i * -0.5
 -> bs::pow(xf, yf) =          -14.5345 + i * 27.4772
 zf =                          0 + i * 0
 -> bs::pow(zf, zf) =          1 + i * 0
 r1f =                         -1
 r2f =                         1.5
 -> bs::pow(r1f, r2f) =        -nan
 -> bs::cmplx::pow(r1f, r2f) = 0 + i * -1
