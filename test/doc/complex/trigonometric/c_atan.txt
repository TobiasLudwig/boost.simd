//! [atan]
---- simd
<- pf =          (1, 2, -1, 0.5) + i * (0, 1, -1, 3)
-> bs::atan(pf) = (0.785398, 1.1781, -1.01722, 1.51048) + i * (-0, 0.173287, -0.402359, 0.335293)
---- scalar
<- xf =                 0 + i * 1
-> bs::atan(xf) =        -nan + i * inf
<- yf =                 -2
-> bs::atan(yf) =        -1.10715
//! [atan]
