//! [sin]
---- simd
<- pf =          (1, 2, -1, 0.5) + i * (0, 1, -1, 3)
-> bs::sin(pf) = (0.841471, 1.40312, -1.29846, 4.82669) + i * (0, -0.489056, -0.634964, 8.79151)
---- scalar
<- xf =                 0 + i * 1
-> bs::sin(xf) =        0 + i * 1.1752
<- yf =                 -2
-> bs::sin(yf) =        -0.909297
