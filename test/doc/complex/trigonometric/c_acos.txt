//! [acos]
---- simd
<- pf =          (1, 2, -1, 0.5) + i * (0, 1, -1, 3)
-> bs::acos(pf) = (0, 0.507356, 2.23704, 1.41379) + i * (0, -1.46935, 1.06128, -1.83019)
---- scalar
<- xf =                 0 + i * 1
-> bs::acos(xf) =        1.5708 + i * -0.881374
<- yf =                 -2
-> bs::acos(yf) =        -nan
//! [acos]
