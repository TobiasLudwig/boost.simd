//! [asin]
---- simd
<- pf =          (1, 2, -1, 0.5) + i * (0, 1, -1, 3)
-> bs::asin(pf) = (1.5708, 1.06344, -0.666239, 0.157005) + i * (0, 1.46935, -1.06128, 1.83019)
---- scalar
<- xf =                 0 + i * 1
-> bs::asin(xf) =        0 + i * 0.881374
<- yf =                 -2
-> bs::asin(yf) =        -nan
-> bs::cmplx::asin(yf) = -1.5708 + i * 1.31696
//! [asin]
