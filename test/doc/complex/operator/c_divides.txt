//! [divides]
---- simd
<- pf =          (1, 2, -1, 0.5) + i * (0, 1, -1, 3)
<- qf =          (3, 1, 1, 0) + i * (0.4, 2, -1, 2)
-> pf - qf     = (0.327511, 0.8, 0, 1.5) + i * (-0.0436681, -0.6, -1, -0.25)
---- scalar
<- xf =          1 + i * 1
<- yf =          1 + i * 1
-> xf - yf     = 1 + i * 0
