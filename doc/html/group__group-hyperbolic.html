Hyperbolic functions
User API » Functions and Constants

Description

These functions provide scalar and SIMD version of hyperbolic and inverse hyperbolic functions.

All these functions can be included individually or all of them just by including <boost/simd/hyperbolic.hpp>

  • Direct hyperbolic functors

    cosh coth csch sech
    sinh sinhc sinhcosh tanh

    sinhc is the hyperbolic sinus cardinal function ( \(\sinh x/x\)).

    sinhcosh allows the simultaneous computation of hyperbolic sine and cosine at lower cost.

    sech and csch are the hyperbolic secant (inverse of the hyperbolic cosine) and cosecant (inverse of the hyperbolic sine)

    Example:
    #include <boost/simd/hyperbolic.hpp>
    #include <boost/simd/pack.hpp>
    #include <boost/simd/function/enumerate.hpp>
    #include <iostream>
    namespace bs = boost::simd;
    using pack_ft = bs::pack <float, 8>;
    int main()
    {
    pack_ft p = bs::enumerate<pack_ft>(-4.0f, 1.0f);
    std::cout << " p = " << p << std::endl
    << " -> bs::cosh(p) = " << bs::cosh(p) << std::endl
    << " -> bs::sinh(p) = " << bs::sinh(p) << std::endl
    << " -> bs::tanh(p) = " << bs::tanh(p) << std::endl
    << " -> bs::sech(p) = " << bs::sech(p) << std::endl
    << " -> bs::csch(p) = " << bs::csch(p) << std::endl;
    pack_ft s, c;
    std::tie(s, c) = bs::sinhcosh(p);
    std::cout << " using sinhcosh " << std::endl
    << " -> bs::cosh(p) = " << c << std::endl
    << " -> bs::sinh(p) = " << s << std::endl;
    return 0;
    }
    Possible output:
    p = (-4, -3, -2, -1, 0, 1, 2, 3)
    -> bs::cosh(p) = ( 27.3082, 10.0677, 3.7622, 1.54308, 1, 1.54308, 3.7622, 10.0677)
    -> bs::sinh(p) = (-27.2899, -10.0179, -3.62686, -1.1752, 0, 1.1752, 3.62686, 10.0179)
    -> bs::tanh(p) = (-0.999329, -0.995055, -0.964028, -0.761594, 0, 0.761594, 0.964028, 0.995055)
    -> bs::sech(p) = (0.036619, 0.0993279, 0.265802, 0.648054, 1, 0.648054, 0.265802, 0.0993279)
    -> bs::csch(p) = (-0.0366436, -0.0998216, -0.275721, -0.850918, inf, 0.850918, 0.275721, 0.0998216)
    using sinhcosh
    -> bs::cosh(p) = (27.3082, 10.0677, 3.7622, 1.54308, 1, 1.54308, 3.7622, 10.0677)
    -> bs::sinh(p) = (-27.2899, -10.0179, -3.62686, -1.1752, 0, 1.1752, 3.62686, 10.0179)

  • Inverse hyperbolic functors

    acosh acoth acsch
    asech asinh atanh
    Example:
    #include <boost/simd/hyperbolic.hpp>
    #include <boost/simd/pack.hpp>
    #include <boost/simd/function/enumerate.hpp>
    #include <iostream>
    namespace bs = boost::simd;
    using pack_ft = bs::pack <float, 8>;
    int main()
    {
    pack_ft p = bs::enumerate<pack_ft>(-2.0f, 0.5f);
    std::cout << " p = " << p << std::endl
    << " -> bs::acosh(p) = " << bs::acosh(p) << std::endl
    << " -> bs::asinh(p) = " << bs::asinh(p) << std::endl
    << " -> bs::atanh(p) = " << bs::atanh(p) << std::endl
    << " -> bs::asech(p) = " << bs::asech(p) << std::endl
    << " -> bs::acsch(p) = " << bs::acsch(p) << std::endl;
    return 0;
    }
    Possible output:
    p = (-2, -1.5, -1, -0.5, 0, 0.5, 1, 1.5)
    -> bs::acosh(p) = (-nan, -nan, -nan, -nan, -nan, -nan, 0, 0.962424)
    -> bs::asinh(p) = (-1.44364, -1.19476, -0.881374, -0.481212, 0, 0.481212, 0.881374, 1.19476)
    -> bs::atanh(p) = (nan, nan, -inf, -0.549306, 0, 0.549306, inf, -nan)
    -> bs::asech(p) = (-nan, -nan, -nan, -nan, inf, 1.31696, 0, -nan)
    -> bs::acsch(p) = (-0.481212, -0.625145, -0.881374, -1.44364, inf, 1.44364, 0.881374, 0.625145)
