This macros allow user-defined types to be dynamically allocated while respecting alignment constraints.

Example:
class custom_type
{
public:
BOOST_SIMD_ALIGNED_OBJECT(custom_type);
private:
boost::simd::pack<char,16> q;
boost::simd::pack<float> p;
};
