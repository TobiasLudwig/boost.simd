Predicates functions

Description

These functions provide scalar and SIMD algorithms for testing properties of values.

Tests such as checks for validity, infinity, sign and others are provided. As for Boolean functions, these functions handle and return typed logical types.

All these functions can be included individually or all of them just by including <boost/simd/predicates.hpp>

  • These predicates can be used through an operator

    name operator name operator
    is_equal == is_less_equal <=
    is_greater_equal >= is_less <
    is_greater > is_not_equal !=

    for floating entries negation as predicates as is_not_greater(a, b) are not equivalent to !is_greater(a, b) due to possible Nan values;

    Example:
    #include <boost/simd/predicates.hpp>
    #include <boost/simd/pack.hpp>
    #include <boost/simd/function/enumerate.hpp>
    #include <boost/simd/constant/inf.hpp>
    #include <boost/simd/constant/minf.hpp>
    #include <boost/simd/constant/nan.hpp>
    #include <boost/simd/function/if_else.hpp>
    #include <iostream>
    namespace bs = boost::simd;
    using pack_ft = bs::pack <float, 8>;
    int main()
    {
    pack_ft p = {0.0f, 1.0f, -1.0f, 2.5f, -2.5f, bs::Nan<float>(), bs::Minf<float>(), bs::Inf<float>()};
    pack_ft q = bs::if_else(bs::is_odd(p), p, p*2.0f);
    std::cout << " p = " << p << std::endl
    << " q = " << q << std::endl
    << " -> (p == q) = " << (p == q) << std::endl
    << " -> (p <= q) = " << (p <= q) << std::endl
    << " -> (p >= q) = " << (p >= q) << std::endl
    << " -> (p < q) = " << (p < q) << std::endl
    << " -> (p > q) = " << (p > q) << std::endl
    << " -> (p != q) = " << (p != q) << std::endl
    << " -> !(p >= q) = " << !(p >= q)<< std::endl;
    return 0;
    }
    Possible output:
    p = (0, 1, -1, 2.5, -2.5, -nan, -inf, inf)
    q = (0, 1, -1, 5, -5, -nan, -inf, inf)
    -> (p == q) = (true, true, true, false, false, false, true, true)
    -> (p <= q) = (true, true, true, true, false, false, true, true)
    -> (p >= q) = (true, true, true, false, true, false, true, true)
    -> (p < q) = (false, false, false, true, false, false, false, false)
    -> (p > q) = (false, false, false, false, true, false, false, false)
    -> (p != q) = (false, false, false, true, true, true, false, false)
    -> !(p >= q) = (false, false, false, true, false, true, false, false)

    Many of these are restriction to special values of some of the preceding ones. Their use often provides speed gain in SIMD mode.

    Example:
    #include <boost/simd/predicates.hpp>
    #include <boost/simd/pack.hpp>
    #include <boost/simd/function/enumerate.hpp>
    #include <boost/simd/constant/minf.hpp>
    #include <boost/simd/constant/nan.hpp>
    #include <boost/simd/constant/mindenormal.hpp>
    #include <iostream>
    namespace bs = boost::simd;
    using pack_ft = bs::pack <float, 8>;
    int main()
    {
    pack_ft p = {0.0f, 1.0f, -1.0f, 2.5f, -2.5f, bs::Nan<float>(), bs::Minf<float>(), bs::Mindenormal<float>()};
    std::cout << " p = " << p << std::endl
    << " -> bs::is_finite(p) = " << bs::is_finite(p) << std::endl
    << " -> bs::is_inf(p) = " << bs::is_inf(p) << std::endl
    << " -> bs::is_denormal(p) = " << bs::is_denormal(p) << std::endl
    << " -> bs::is_flint(p) = " << bs::is_flint(p) << std::endl
    << " -> bs::is_odd(p) = " << bs::is_odd(p) << std::endl
    << " -> bs::is_nan(p) = " << bs::is_nan(p) << std::endl;
    return 0;
    }
    Possible output:
    p = (0, 1, -1, 2.5, -2.5, -nan, -inf, 1.4013e-45)
    -> bs::is_finite(p) = (true, true, true, true, true, false, false, true)
    -> bs::is_inf(p) = (false, false, false, false, false, false, true, false)
    -> bs::is_denormal(p) = (false, false, false, false, false, false, false, true)
    -> bs::is_flint(p) = (true, true, true, false, false, false, false, false)
    -> bs::is_odd(p) = (false, true, true, false, false, false, false, false)
    -> bs::is_nan(p) = (false, false, false, false, false, true, false, false)
